## | 自动加载文件夹下 js 文件

```javascript
// 自动加载router文件夹下的所有js文件
const context = require.context('@/router', false, /\.js$/);
let routesConfig = [];
context.keys().forEach((v) => {
  if (v === './index.js') return;
  let obj = context(v);
  routesConfig = routesConfig.concat(obj[Object.keys(obj)[0]]);
});
```
